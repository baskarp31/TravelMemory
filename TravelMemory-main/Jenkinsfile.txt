pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/baskarp31/TravelMemory.git'
            }
        }
        stage('Install') {
            steps {
                sh 'cd TravelMemory-main\backend; npm install'
            }
        }
        stage('Build') {
            steps {
                sh 'cd TravelMemory-main\backend; npm run build'
            }
        }
    }

}
